# 集合

  • 一种无序且唯一的数据结构

  • ES6中有集合, 名为Set

  • 集合常用操作: 去重,判断某元素是否在集合中,求交集

    entries(): 返回一个迭代器，值为键值对

    values(): 返回一个迭代器, 值为集合的值

    keys(): 返回一个迭代器，值为集合中的所有键

# 字典

  • 与集合相似, 字典也是一种存储唯一值的数据结构, 但他是以键值对的形式存储
  • ES6中有字典-->Map(映射)
  • 常见操作 增(set) 删(delete) 改(set) 查(get)

# 后序遍历算法

  1. 对根节点的左子树进行中序遍历

  3. 访问根接节点
//递归
const postorder = (root) => {
  if (!root) return;
  postorder(root.left);
  postorder(root.right);
  console.log(root.val);
};
postorder(binaryTree);

//非递归
const inorder = root => {
  if (!root) return;
  const outputStack = [];
  const stack = [root];
  while (stack.length) {
    const n = stack.pop();
    outputStack.push(n);
    if (n.left) stack.push(n.left);
    if (n.right) stack.push(n.right);
  }
  while (outputStack.length) {
    const n = outputStack.pop();
    console.log(n.val);
  }
}
inorder(binaryTree)

# 513.找树左下角的值

var findBottomLeftValue = function(root) {
    let maxPath = 0 //最大深度
    let res = 0
    const dfsTree = function(node,curPath){
        // 递归终止条件
        if(node.left === null && node.right===null){
            if(curPath > maxPath){
                maxPath = curPath
                res = node.val
            }
            return
        }
        node.left && dfsTree(node.left , curPath + 1)
        node.right && dfsTree(node.right , curPath + 1)
    }
    dfsTree(root,1)
    return res
};
//迭代 -层序遍历
var findBottomLeftValue = function(root) {
    let queue = []
    if(root===null) return 0 
    queue.push(root)
    let res = 0
    while(queue.length){
        let len =queue.length
        for(let i=0 ; i< len ;i++){
            let node = queue.shift()
            if(i === 0) res = node.val
            node.left && queue.push(node.left)
            node.right && queue.push(node.right)
        }
    }
    return res
};

# 113.路径总和 II

// 113. 路径总和 II  把遍历树的值保存--可作为模板使用 递归
var pathSum = function(root, targetSum) {
    let res = []
    const dfs = function(node,target,curVal){
        if(!node) return 
        // 终止条件
        if(!node.left && !node.right && target === node.val) {
            curVal.push(node.val)
            res.push([...curVal])
            curVal.pop()
            return 
        }
        curVal.push(node.val)
        node.left && dfs(node.left,target -node.val,curVal)
        node.right && dfs(node.right,target -node.val,curVal)
        curVal.pop()
    }
    dfs(root,targetSum,[])
    return res
};

//迭代
let hasPathSum = function(root, targetSum) {
    if(root === null) return false;
    let nodeArr = [root];
    let valArr = [0];
    while(nodeArr.length) {
        let curNode = nodeArr.shift();
        let curVal = valArr.shift();
        curVal += curNode.val;
        // 为叶子结点，且和等于目标数，返回true
        if (curNode.left === null && curNode.right === null && curVal === targetSum) {
            return true;
        }
        // 左节点，将当前的数值也对应记录下来
        if (curNode.left) {
            nodeArr.push(curNode.left);
            valArr.push(curVal);
        }
        // 右节点，将当前的数值也对应记录下来
        if (curNode.right) {
            nodeArr.push(curNode.right);
            valArr.push(curVal);
        }
    }
    return false;
};

//正确写法
var combinationSum = function(candidates, target) {
    let res = []
    let path = []
    candidates.sort()
    let sum = 0
    function backtracking(candidates,target, startIndex){
        // 终止条件写在外面
        if(sum > target) {
            return

            // res.push(path.slice())
            res.push([...path])
            return

            path.push(candidates[i])

            sum -= candidates[i]
        }

# 491. 递增子序列

var findSubsequences = function(nums) {
    let res = [] ,path = []
    let set = new Set()

    function backtracking(nums,startIndex){
        if (path.length > 1 && !set.has(path.toString())) {
            set.add(path.toString())
            res.push([...path])
        }
        for(let i = startIndex ; i< nums.length ;i++){
            if(path.length == 0 || path[path.length - 1] <= nums[i]){
                path.push(nums[i])
                backtracking(nums,i+1)
                path.pop()
            }
        }
    }
    backtracking(nums,0)
    return res
};

var largestSumAfterKNegations = function(nums, k) {
    // 按绝对值从大到小排列，然后遍历遇到nums[i]<0就反转，结束后如果k还不等于0，就最后一个数不断反转耗尽k值
    nums.sort((a,b)=>{
        return Math.abs(b) - Math.abs(a)
    })
    for(let i =0;i < nums.length;i++){
        if(nums[i] < 0 && k >0){
            nums[i] *= -1
            k--
        }
    }
    while(k){
        nums[nums.length - 1] *= -1
        k--
    }
    return nums.reduce((pre,cur)=>{
        return cur + pre
    })
};

    let res = 0
    let minPrice = prices[0]
    for(let i = 1; i < prices.length;i++){
        // 情况二：相当于买入
        if(prices[i] < minPrice){
            minPrice = prices[i]
        }
        // 情况三：保持原有状态（因为此时买则不便宜，卖则亏本）
        if(prices[i] >= minPrice && prices[i] <= minPrice + fee){
            continue
        }

        }
    }
    return res
};

# 背包问题 掌握01背包和完全背包就完全够用了

01背包理论
dp[i][j] 表示从下标为[0-i]的物品里任意取，放进容量为j的背包，价值总和最大是多少。
一、初始化
dp[i][0] = 0 因为背包重量为0无法放东西
dp[0][j]: if(j < weight[i]) dp[0][j] = 0 
		else if(j >= weight[i]) dp[0][j] = value[0]
其余数值均初始化为0
二、递推公式
	不放物品i，那么dp[i][j] 就是 dp[i-1][j]即上一个物品的价值
	放物品i，dp[i][j] = max(dp[i-1][j],dp[i-1][j-weight[i]] + value[i])

先遍历物品再遍历背包
dp[j - weight[i]]表示容量为j - weight[i]的背包所背的最大价值。
dp[j - weight[i]] + value[i] 表示 容量为 j - 物品i重量 的背包 加上 物品i的价值。（也就是容量为j的背包，放入物品i了之后的价值即：dp[j]）

    // 背包如果正好装满，说明找到了总和为 sum / 2 的子集。

    if (sum & 1) return false;
    // 背包的体积为sum / 2
    const dp = Array(sum / 2 + 1).fill(0)
    // 倒叙遍历，先遍历物品，再遍历背包容量
    for(let i = 0;i <= nums.length -1 ;i++){
        for(let j = sum/2;j >= nums[i];j--){
            dp[j] = Math.max(dp[j],dp[j-nums[i]] + nums[i])
        }
    }
    return dp[sum / 2] === sum / 2
};

# 1049. 最后一块石头的重量 II

var lastStoneWeightII = function(stones) {
    //本题其实就是尽量让石头分成重量相同的两堆，相撞之后剩下的石头最小，这样就化解成01背包问题了
    // dp[j] 表示背包容量为j
    let sum = stones.reduce((a,b) => a+b) 
    let target = Math.floor(sum/2)
    const dp = new Array(target + 1).fill(0)
    //先遍历物品再遍历重量    
    for(let i = 0;i<stones.length;i++){
        for(let j = target; j >= stones[i];j--){
            dp[j] = Math.max(dp[j],dp[j - stones[i]] + stones[i])
        }
    }
    return sum - dp[target] - dp[target];
};

# 494. 目标和

var findTargetSumWays = function(nums, target) {
    // x为加和，sum-x为减和,target = x - (sum-x),所以x=(target+sum)/2
    // 思维：以一种变量进行dp
    // dp[j] 表示装入容量为j的背包的方案有多少种
    // 递推公式：不考虑nums[i]的情况下，填满容量为j - nums[i]的背包，有dp[j - nums[i]]种方法。
    let sum = nums.reduce((a,b)=>a+b)
    // ①target的绝对值比sum大，无解②加和不能为小数
    if(Math.abs(target) > sum) {
        return 0;
    }
    if((target + sum)%2 === 1) return 0

    let x = (target + sum)/2
    let dp = new Array(x+1).fill(0)
    dp[0] = 1
    for(let i = 0 ;i < nums.length;i++){
        for(let j = x ; j >= nums[i] ;j-- ){
            dp[j] += dp[j - nums[i]]
        }
    }

    return dp[x]

};
