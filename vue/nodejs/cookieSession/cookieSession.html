# 解析cookie

const serverHandler = (req,res) =>{
    // 设置响应格式
    res.setHeader('Content-Type','application/json')
    // /user/login?username=zhangsan&password=123   -> url
    // /user/login                                  -> path
    // username=zhangsan&password=123               ->查询参数 query
    const url = req.url
    const method = req.method
    // req本身没有path和query是新增的
    req.path = url.split('?')[0]
    req.query = querystring.parse(url.split('?')[1])

    console.log(req.headers.cookie)
    //解析cookie
    //cookie:name=zhangsan; age=20
    req.cookie = {}
    const cookieStr = req.headers.cookie || ''
    cookieStr.split(';').forEach(item => {
        if(!item) return
        const arr = item.split('=')
        const key = arr[0].trim()
        const val = arr[1].trim()
        req.cookie[key] = val
    });
    
    console.log('path',req.path);
    console.log('query',req.query);
    if(method === 'GET' && url === '/user/login'){
        res.end("你好")
        return
    }
    res.end(JSON.stringify(req.cookie))
}

# 登录验证和设置cookie

--src ---routes ----user.js

const { login } = require('../controllers/user')

// 设置cookie过期时间
const setCookieExpiresTime = function(){
    const date = new Date()
    date.setTime(date.getTime() + 10 * 1000)
    return date.toGMTString()
}

const handleUserRoute = (req,res) =>{
    const method = req.method
    if(method === 'GET' && req.path === '/api/user/login'){
        // http://localhost:3000/api/user/login?username=zhangsan&password=123456
        const { username, password } = req.query
        const loginRes = login(username,password)
        // 如果用户名密码正确，设置cookie并且返回结果
        if(loginRes){
            res.setHeader('Set-Cookie',`username=${username}; path=/; expires=${setCookieExpiresTime()}`)
            return {
                errorCode:0,
                data: req.query
            }
        }
        return {
            errorCode:-1,
            data: req.query
        }
    }
}
module.exports = handleUserRoute
